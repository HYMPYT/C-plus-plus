int sum_digits(int a)
{
	if (a < 10)
		return a;
	return a % 10 + sum_digits(a / 10);
}
int factorial_r(int val)
{
	if (val < 2)
		return 1;
	return val * factorial_r(val - 1);
}
int factorial(int val)
{
	int f = 1;
	while (val)
		f *= val--;
	return f;
}
int digits_counter(int a)
{
	if (a < 10)
		return 1;
	return 1 + digits_counter(a / 10);
}
int Sum(int a, int b, int c)
